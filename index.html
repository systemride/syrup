Syrup

Syrup is a collection of shared UI utilities and libraries leveraged by AI2 when developing interfaces.

It is opinionated, simple and based upon LESS and AngularJS

Installation:

npm install syrup

Usage:

Gulp

var gulp = require('gulp');
var syrup = require('syrup');
syrup.gulp.tasks(
  gulp,
  {
    base: __dirname,
    html: 'src/**/*.html',
    less: 'src/styles.less',
    js: 'src/main.js',
    assets: 'src/assets/**/*',
    build: 'build',
    tmp: 'tmp',
    watch: 'src/',
    unitTests: 'src/**/*-test.js',
    unitTestConfig: 'karma.conf.js',
    integrationTestConfig: 'protractor.conf.js'
  },
  'dev',
);
